### Openssl hashing (create a hash of file)

# Symmetric Encryption ----->
openssl enc -aes-256-cbc -salt -in file.txt -out file.enc
openssl enc -aes-256-cbc -d -in file.enc -out decrypted.txt

# Asymmetric Encryption ----->
openssl genrsa -out private.key 2048
openssl rsa -in private.key -pubout -out public.key

openssl pkeyutl -encrypt -pubin -inkey public.key -in file.txt -out encrypted.bin
openssl pkeyutl -decrypt -inkey private.key -in encrypted.bin -out decrypted.txt

# Hashing ----->
openssl dgst -sha256 file.txt

openssl dgst -sha256 file.txt > hash.txt

# SHA-256 Hash
openssl dgst -sha256 file.txt

# SHA-1 Hash
openssl dgst -sha1 file.txt

# MD5 Hash
openssl dgst -md5 file.txt

# SHA-512 Hash
openssl dgst -sha512 file.txt

# Hash save in file 
openssl dgst -sha256 file.txt > hash.txt

# only show hash value
openssl dgst -sha256 -r file.txt

# Multiple files  hash
openssl dgst -sha256 file1.txt file2.txt file3.txt

# PHASE 1 - ROOT CA


mkdir rootca
cd rootca
mkdir certs csr private

openssl genrsa -out private/rootCA11.key 4096

openssl req -x509 -new \
-key private/rootCA11.key \
-sha256 \
-days 3650 \
-out certs/rootCA11.crt

openssl x509 -in certs/rootCA11.crt -text -noout


# 
# PHASE 2 - SUB CA
# 

cd ..
mkdir subca
cd subca
mkdir certs csr private

openssl genrsa -out private/subCA11.key 4096

openssl req -new \
-key private/subCA11.key \
-out csr/subCA11.csr

nano subca_ext.cnf

# Paste below:
# basicConstraints=critical,CA:TRUE,pathlen:0
# keyUsage=critical,keyCertSign,cRLSign
# subjectKeyIdentifier=hash
# authorityKeyIdentifier=keyid,issuer

openssl x509 -req \
-in csr/subCA11.csr \
-CA ../rootca/certs/rootCA11.crt \
-CAkey ../rootca/private/rootCA11.key \
-CAcreateserial \
-out certs/subCA11.crt \
-days 1825 \
-sha256 \
-extfile subca_ext.cnf

openssl verify \
-CAfile ../rootca/certs/rootCA11.crt \
certs/subCA11.crt

# PHASE 3 - WEBSITE CERTIFICATE
# 

cd ..
mkdir website
cd website
mkdir certs csr private

openssl genrsa -out private/boomshakalaka.key 2048

openssl req -new \
-key private/boomshakalaka.key \
-out csr/boomshakalaka.csr

nano server_ext.cnf

# Paste below:
# basicConstraints=critical,CA:FALSE
# keyUsage=critical,digitalSignature,nonRepudiation,keyEncipherment,keyAgreement
# extendedKeyUsage=serverAuth,clientAuth,emailProtection
# subjectAltName=@alt_names
#
# [alt_names]
# DNS.1=www.boomshakalaka.in

openssl x509 -req \
-in csr/boomshakalaka.csr \
-CA ../subca/certs/subCA11.crt \
-CAkey ../subca/private/subCA11.key \
-CAcreateserial \
-out certs/boomshakalaka.crt \
-days 365 \
-sha256 \
-extfile server_ext.cnf

openssl x509 -in certs/boomshakalaka.crt -text -noout

$$$$ XCA HTTPS WEBSITE USING ROOT CA → SUB CA → END ENTITY ------

1) ROOT CA CREATE
-----------------
XCA → Private Keys → New Key
Algorithm: RSA
Key Size: 4096

Certificates → New Certificate

Source:
  Create Self-Signed Certificate

Subject:
  CN = RootCA

Extensions:
  Basic Constraints = CA:TRUE
  Key Usage = Certificate Sign, CRL Sign

Save as RootCA

------------------------------------------------------------

2) SUB CA CREATE
----------------
Private Keys → New Key
RSA 4096

Certificates → New Certificate

Source:
  Signer = RootCA

Subject:
  CN = SubCA

Extensions:
  Basic Constraints = CA:TRUE
  Key Usage = Certificate Sign, CRL Sign

Save as SubCA

------------------------------------------------------------

3) END ENTITY / SERVER CERTIFICATE
----------------------------------
Private Keys → New Key
RSA 2048

Certificates → New Certificate

Source:
  Signer = SubCA

Subject:
  CN = localhost
  (or server IP / hostname)

Extensions:
  Basic Constraints = CA:FALSE
  Key Usage = Digital Signature, Key Encipherment
  Extended Key Usage = Server Authentication

Save as ServerCert

------------------------------------------------------------

4) EXPORT FILES
---------------
RootCA      → rootca.crt
SubCA       → subca.crt
ServerCert  → server.crt
Server Key  → server.key

------------------------------------------------------------

-----------------
sudo apt update
sudo apt install apache2 -y
sudo a2enmod ssl
